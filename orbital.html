import math

def get_delta_v(r1, r2):
    """
    Calculates Delta-V for Hohmann Transfer.
    r1: Initial orbit radius (km)
    r2: Target orbit radius (km)
    mu: Earth's gravitational parameter
    """
    mu = 3.986004418e5 
    
    # Phase 1: Injection burn
    v1 = math.sqrt(mu / r1)
    v_perigee = math.sqrt(mu / r1) * math.sqrt((2 * r2) / (r1 + r2))
    dv1 = v_perigee - v1
    
    # Phase 2: Circularization burn
    v2 = math.sqrt(mu / r2)
    v_apogee = math.sqrt(mu / r2) * math.sqrt((2 * r1) / (r1 + r2))
    dv2 = v2 - v_apogee
    
    return dv1 + dv2

# Example: LEO to Geostationary
total_burn = get_delta_v(6700, 42164)
print(f"Total Delta-V: {total_burn:.4f} km/s")

double calculateDeltaV(double r1, double r2) {
    const double mu = 398600.44;
    
    double v1 = std::sqrt(mu / r1);
    double v_perigee = v1 * std::sqrt((2 * r2) / (r1 + r2));
    double dv1 = v_perigee - v1;
    
    double v2 = std::sqrt(mu / r2);
    double v_apogee = v2 * std::sqrt((2 * r1) / (r1 + r2));
    double dv2 = v2 - v_apogee
    
    return dv1 + dv2;
}

Δv = √ (μ / r₁) * [ √ (2r₂ / (r₁ + r₂)) - 1 ]

The Hohmann Transfer Orbit uses two impulsive burns to move a spacecraft between two coplanar circular orbits.

μ (Mu): Standard gravitational parameter.
r₁: Starting radius (Earth Orbit).
r₂: Destination radius (Target Orbit).
